## Installation

```bash
pnpm dlx shadcn@latest add @sylph-ui/page
```

## Usage

```tsx
import {
  Page,
  PageAction,
  PageContent,
  PageDescription,
  PageHeader,
  PageTitle,
} from "@repo/page";
```

```tsx
<Page>
  <PageHeader>
    <PageTitle>Products</PageTitle>
    <PageDescription>Manage your products and inventory.</PageDescription>
  </PageHeader>
  <PageContent>{/* Page content */}</PageContent>
</Page>
```
